#!/usr/bin/env python

# Carry out specific functions when asked to by the system
case "$1" in
  start)
    python /home/pi/Desktop/eds_azzzist/solo_ys/april/apr9/ON_CHIME.py
    echo "Starting listen-for-shutdown.py"
    /usr/local/bin/listen-for-shutdown.py &
